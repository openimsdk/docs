### 调用示例

```js showLineNumbers
import { getSDK } from 'open-im-sdk-wasm';
import { CbEvents } from 'open-im-sdk-wasm/constant';
const IMSDK = getSDK();

IMSDK.on(
  CbEvents.OnNewConversation,
  ({ data }: WSEvent<ConversationItem[]>) => {
    // data 会话信息
  }
);
```
